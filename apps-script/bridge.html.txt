<body>
<script>
(() => {
  'use strict';

  const ALLOWED_ORIGINS = [
    'https://portalsimonsports.github.io'
  ];

  function isAllowed(origin) {
    return ALLOWED_ORIGINS.includes(String(origin || ''));
  }

  function send(target, origin, data) {
    try {
      target.postMessage(data, origin);
    } catch (_) {}
  }

  window.addEventListener('message', event => {
    if (!isAllowed(event.origin)) return;

    const data = event.data || {};
    if (data.type !== 'PSS_BRIDGE_CALL' || !data.id) return;

    const id = String(data.id);
    const request = data.request || {};

    google.script.run
      .withSuccessHandler(result => {
        send(event.source, event.origin, {
          type: 'PSS_BRIDGE_RESPONSE',
          id,
          ok: true,
          result
        });
      })
      .withFailureHandler(error => {
        send(event.source, event.origin, {
          type: 'PSS_BRIDGE_RESPONSE',
          id,
          ok: false,
          error: String(error && error.message ? error.message : error || 'Falha na API.')
        });
      })
      .PSS_PESAGEM_bridgeCall(request);
  });

  window.parent.postMessage({ type: 'PSS_BRIDGE_READY' }, '*');
})();
</script>
</body>
